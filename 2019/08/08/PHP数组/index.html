创建数组

$arr = array()

$arr = array(1,2,3,4,5)

$arr = array('name' => 'zs', 'age' => 18)
// 下边的方法，只有在php5.4之后才可以

$arr = []

$a = '123';

索引数组

赋值

// 创建时直接赋值

$arr = array(1,2,3,4,5)

$arr = [1, 2, 3, 4]

// 先创建数组

$arr = array(1,2,3,4,5)

$arr = array()

$arr[0] = 123   // ==>  [123, 2, 3, 4, 5]

$arr[100] = 456  // ==> [1, 2, 3, 4, 5,..., 456]

取值

// 通过下标取值

$arr = array(1,2,3,4,5)

echo $arr[4]

// 通过循环遍历

for($i = 0; $i < 5; $i ++ ){    // 为什么从0开始，是因为数组中，下标是从0开始的
    echo $arr[$i];
}  

for($i = 0; $i < count($arr); $i ++ ){
    echo $arr[$i];
} 

foreach($arr as $key => $value){
    echo $key . '' . $value;
}

foreach($arr as $value){
    echo $value;
}

数组常用函数

array_keys( $arr  ) // 返回一个索引数组   包含所有数组的键

array_values( $arr  ) // 返回一个索引数组  包含所有数组的值

count()  sizeof()

array_merge() // 合并数组   相当于 +

array_key_exists() // 判断键是否存在

sort() // 排序

in_array()  // 判断元素是否在数组中

array_unique() // 去重

array_pop()  // 删除最后一个元素
